/**
 * 
 * @class Timeline
 * @description
 *
 * Object representing a timeline instance
 *
 * @constructor
 * @param {string} timelineId Unique identifier for this timeline
 * @param {string} timelineType Timeline type specifier
 * @param {string} contentId Unique identifier for this timeline
 * @param {CorrelatedClock} [clock=null]
 * @param {string} [updateChannel=null]
 */
var Timeline = function(timelineId, timelineType, contentId, clock, updateChannel) {

    if ((typeof timelineType !== "string") || 
        (typeof timelineId !== "string") ||
        (typeof contentId !== "string") ) {
        throw ("Missing one or more mandatory parameters or saw one or more wrong parameter types.");
    }

    Object.defineProperties(this, {
        "timelineId": {
            enumerable: true,
            value: timelineId
        },
        "timelineType": { 
            enumerable: true, 
            value: timelineType
        },
        "contentId": {
            enumerable: true,
            value: contentId
        },
        "clock": {
            enumerable: true,
            writable: true,
            value: clock || null
        },
        "updateChannel": {
            enumerable: true,
            writable: true,
            value: updateChannel || null
        }
    });
};

module.exports = Timeline;
