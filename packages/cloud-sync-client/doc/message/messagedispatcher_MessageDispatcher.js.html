var WeakMap = require("weak-map"),
    PRIVATE = new WeakMap(),
    MessageDispatcher;

/**
 * @constructor
 * @name MessageDispatcher
 */
MessageDispatcher = function () {
    var priv;
    PRIVATE.set(this, {});
    priv = PRIVATE.get(this);
    priv.messageMap = {};
};

/**
 * Sets a callback for a given message. The callback is invoked, if a message 
 * with the same 'id' is passed to method 'call'.
 * 
 * @memberof MessageDispatcher
 * @param {Message} message
 * @param {function} handler Callback function to be called. In this function
 *      this refers to the message that invoked its call, i.e. the message that
 *      is passed to method 'call'.
 */
MessageDispatcher.prototype.set = function (message, handler) {
    var priv, typeOfHandler;
    priv = PRIVATE.get(this);
    typeOfHandler = typeof handler;
    if (typeOfHandler !== "function") {
        throw new Error("Expected 'handler' to be of type function. Saw '" + typeOfHandler + "' instead");
    }
    priv.messageMap[message.id] = handler;
};

/**
 * Invokes the call of the handler set with method 'set', if the 'id' property
 * value of the message passed to call equals the 'id' of a previousely set
 * handler.
 * 
 * @memberof MessageDispatcher
 * @param {Message} message
 */
MessageDispatcher.prototype.call = function (message) {
    var priv = PRIVATE.get(this);
    if (priv.messageMap.hasOwnProperty(message.id)) {
        priv.messageMap[message.id].call(message);
    }
};

module.exports = MessageDispatcher;
